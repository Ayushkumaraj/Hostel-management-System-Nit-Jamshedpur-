let complaints = JSON.parse(localStorage.getItem("complaints")) || [];

let currentUser = null;

function login() {

  const id = document.getElementById("login-id").value.trim();

  const password = document.getElementById("login-password").value;

  const error = document.getElementById("login-error");

  // Student login rule: ID must match 2024UGPI### or 2024UGME###

  const studentPattern = /^(2024UGPI\d{3}|2024UGME\d{3})$/;

  if (id === "W001" && password === "warden123") {

    currentUser = { role: "warden", id };

    document.getElementById("login-section").classList.add("hidden");

    document.getElementById("warden-section").classList.remove("hidden");

    renderWardenComplaints();

  } else if (studentPattern.test(id) && password === "student123") {

    currentUser = { role: "student", id };

    document.getElementById("login-section").classList.add("hidden");

    document.getElementById("student-section").classList.remove("hidden");

    renderStudentComplaints();

  } else {

    error.textContent = "Invalid ID or password";

  }

}

function submitComplaint() {

  const name = document.getElementById("student-name").value.trim();

  const room = document.getElementById("room-no").value.trim();

  const block = document.getElementById("block").value.trim();

  const type = document.getElementById("complaint-type").value;

  const manual = document.getElementById("manual-type").value.trim();

  const details = document.getElementById("complaint-details").value.trim();

  if (!name || !room || !block || (!type && !manual) || !details) {

    alert("Please fill all fields!");

    return;

  }

  const complaint = {

    id: complaints.length + 1,

    name,

    room,

    block,

    type: type === "Other" ? manual : type,

    details,

    status: "Pending"

  };

  complaints.push(complaint);

  localStorage.setItem("complaints", JSON.stringify(complaints));

  renderStudentComplaints();

  renderWardenComplaints();

  document.getElementById("student-name").value = "";

  document.getElementById("room-no").value = "";

  document.getElementById("block").value = "";

  document.getElementById("complaint-type").value = "";

  document.getElementById("manual-type").value = "";

  document.getElementById("complaint-details").value = "";

}

function renderStudentComplaints() {

  const tbody = document.getElementById("student-complaints");

  tbody.innerHTML = "";

  complaints.forEach(c => {

    if (currentUser && currentUser.role === "student") {

      const row = `<tr>

        <td>${c.id}</td>

        <td>${c.name}</td>

        <td>${c.room}</td>

        <td>${c.block}</td>

        <td>${c.type}</td>

        <td>${c.details}</td>

        <td>${c.status}</td>

      </tr>`;

      tbody.innerHTML += row;

    }

  });

}

function renderWardenComplaints() {

  const tbody = document.getElementById("warden-complaints");

  tbody.innerHTML = "";

  complaints.forEach((c, index) => {

    const row = `<tr>

      <td>${c.id}</td>

      <td>${c.name}</td>

      <td>${c.room}</td>

      <td>${c.block}</td>

      <td>${c.type}</td>

      <td>${c.details}</td>

      <td>${c.status}</td>

      <td>

        <button onclick="updateStatus(${index}, 'In Progress')">In Progress</button>

        <button onclick="updateStatus(${index}, 'Resolved')">Resolved</button>

      </td>

    </tr>`;

    tbody.innerHTML += row;

  });

}

function updateStatus(index, status) {

  complaints[index].status = status;

  localStorage.setItem("complaints", JSON.stringify(complaints));

  renderWardenComplaints();

}
